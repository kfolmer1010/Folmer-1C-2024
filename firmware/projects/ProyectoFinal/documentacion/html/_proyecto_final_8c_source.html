180void ReadValueTask()
181{
182 while(true)

184 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);

187 if(valor>200&&on==true)
188 {
189 if(conc_alcohol<=200)
190 {
191 tapaOpen = true;
192 TapaMove();
193 alarma = false;
194

196 else if(conc_alcohol>200)
197 {
198 tapaOpen = false;
199 TapaMove();
200 alarma = true;
201
202 }
203 contador += conc_alcohol;
204 var ++;
205 }
206 else
207 {
208 contador = 0;
209 var = 0;
210 }
211 }
